      ## Expresiones extra en IPython
        ?  # añadido tras un comando, muestra doc/ayuda
        ??  # similar, muestra doc/ayuda y el cód. fuente
        In[n] / In[-1] / Out[n]  # Ojo, Out es diccionario!
        _ / __  # último / penúltimo (etc.) resultado - Out!
        ;  # para separar sentencias dentro de una línea
        ! # system command escape (invocar al shell del sistema)

      ## Magic functions
        magic  # equivalente a help() para las magic functions
        %quickref  # ayuda resumida para las magic functions y más
        %lsmagic  # lista todas las magic functions
        %rep  # toma el último resultado (Out) para editar
        %cd
        %pwd  # mostrar el directorio actual

        %edit [-p]  # para usar un editor (opc. previo cont.)
        %run [-i|-m]  # ejec. fichero (opc. "integrado") / módulo
        %cpaste  # útil para pegar código copiado, porque
                 # se permiten espacios iniciales, ">", etc.
        %time ...  # medir el tiempo de ejec. de una sentencia
        %%time ...  # medir el tiempo de ejec. de una celda
        %matplotlib inline  # si usamos matplotlib, ver los result.!
